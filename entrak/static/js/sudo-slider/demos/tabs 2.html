Usage

Include the javascripts

<script type="Text/Javascript" src="http://code.jquery.com/jquery-1.11.0.min.js"></script>
<script type="text/javascript" src="../js/jquery.sudoSlider.min.js"></script>

The Javascript to start it.

<script type="text/javascript" >
   $(document).ready(function(){
      var oldt = 0;
      var sudoSlider = $("#slider").sudoSlider({
         effect: "stackDown",
         speed: 800,
         ajax: [
            '../images/01.jpg',
            '../images/02.jpg',
            '../images/03.jpg',
            '../images/04.jpg',
            '../images/05.jpg'
         ],
         prevNext:true,
         beforeAnimation: function(t){
            var scroll = -t+2;
            if (scroll == 1) scroll = 0;
            if (scroll < -2) scroll = -2;
            var scroll = scroll * $('.custom').eq(0).outerHeight(true);
            var speed = 800;
            $('#slidemenu ul').animate(
               { marginTop: scroll },
               {
                  queue:false,
                  duration:speed
               }
            );
            var substract = $('#slidemenu ul').offset();
            var position = $('#slidemenu ul li').eq(t-1).offset();
            var marginTop =  position.top - substract.top;
            $('#slidemenu ul li.currentone').animate(
               { marginTop: marginTop },
               {
                  queue:false,
                  duration:speed
               }
            );
         },
         customLink: '.custom',
         continuous:true,
         preloadAjax:true
      });
   });
</script>
